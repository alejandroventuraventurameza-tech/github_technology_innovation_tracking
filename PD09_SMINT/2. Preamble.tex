% Redefinir los entornos bluebox, greenbox y redbox con contadores automáticos
\newenvironment{bluebox}[1]
{%
  \stepcounter{defcounter}% Incrementar el contador de definiciones
  \setbeamercolor{block title}{bg=myblue, fg=black}%
  \begin{block}{Definición \thedefcounter\ (#1)}%
}
{\end{block}}

\newenvironment{greenbox}[1]
{%
  \stepcounter{theocounter}% Incrementar el contador de teoremas
  \setbeamercolor{block title}{bg=mygreen, fg=black}%
  \begin{block}{Teorema \thetheocounter\ (#1)}%
}
{\end{block}}

\newenvironment{redbox}[1]
{%
  \stepcounter{excounter}% Incrementar el contador de ejemplos
  \setbeamercolor{block title}{bg=myred, fg=black}%
  \begin{block}{Ejemplo \theexcounter\ (#1)}%
}
{\end{block}}
