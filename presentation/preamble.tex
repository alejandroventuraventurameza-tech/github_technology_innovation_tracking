% Redefinir los entornos bluebox, greenbox y redbox
\newenvironment{bluebox}[1]
{%
  \stepcounter{defcounter}
  \setbeamercolor{block title}{bg=myblue, fg=black}%
  \begin{block}{Definition \thedefcounter\ (#1)}%
}
{\end{block}}

\newenvironment{greenbox}[1]
{%
  \stepcounter{theocounter}
  \setbeamercolor{block title}{bg=mygreen, fg=black}%
  \begin{block}{Result \thetheocounter\ (#1)}%
}
{\end{block}}

\newenvironment{redbox}[1]
{%
  \stepcounter{excounter}
  \setbeamercolor{block title}{bg=myred, fg=black}%
  \begin{block}{Finding \theexcounter\ (#1)}%
}
{\end{block}}
